import java.util.Stack;

import com.hhu.util.Visualizer;

public class StateOfStructure {

    public void main() {
        Stack<Integer> stack = new Stack<>();
        Visualizer.register(stack);
        for(int i = 0; i<4;i++){
        	stack.push(i);

        }
        for(int i = 0; i<4;i++){
        	stack.pop();
            
        }
    }
}
